@startuml
autonumber
actor "Administrator" as ADMIN

participant ":ParameterUI" as UI
participant ":ParameterController" as CTRL
participant ":Company" as CMP
participant "store :ParameterStore" as STORE
participant "p:Parameter" as TET

activate ADMIN
ADMIN -> UI : specifies a new parameter and categorize it
activate UI
UI --> ADMIN : request data (code, shortname, description) and parametercategory
deactivate UI

ADMIN -> UI : types/selects requested data
activate UI
UI -> CTRL: createParameter(code, shortname, description)
activate CTRL
CTRL -> CMP : store = getParameterStore()
deactivate CMP
CTRL -> STORE : P = createParameter (code, shortname, description)
deactivate 
 STORE --> TET** : create(code,shortname,description)
deactivate CMP
CTRL -> STORE : validateParameter(p)
activate CMP
deactivate CMP
deactivate CTRL

UI --> ADMIN : shows the data and request a confirmation
deactivate UI

ADMIN -> UI : confirms the data
activate UI

UI -> CTRL : saveParameter()
activate CTRL

CTRL -> STORE : saveParameter(p)
activate CMP

STORE -> STORE++ : validateParameter(p)
CMP--
STORE -> STORE++ : save.addParameter(p)
CMP--
deactivate CMP
deactivate CTRL
UI --> ADMIN : informs operation succecs
deactivate UI

deactivate ADMIN
@enduml
